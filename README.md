1.Arthimatic operators 

#include<iostream>
using namespace std;
int main()
{
	int a,b;
	cin>>a>>b;
	cout<<"sum is:"<<(a+b)<<endl<<"difference is:"<<(a-b)<<endl<<"product is:"<<(a*b)<<endl<<"quotient is:"<<(a/b)<<endl<<"remainder is:"<<(a%b)<<endl<<"successor is:"<<(++a)<<endl<<"predecessor is:"<<(--b)<<endl;
	return 0;
}
